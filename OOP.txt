public class AutomaticBike {
    private boolean isOn;
    private int speed;
    private int gear;

    public AutomaticBike() {
        this.isOn = false;
        this.speed = 0;
        this.gear = 1;
    }

    public void turnOn() {
        isOn = true;
    }

    public void turnOff() {
        isOn = false;
        speed = 0;  // Optionally reset speed
        gear = 1;   // Reset gear
    }

    public boolean isOn() {
        return isOn;
    }

    public int getSpeed() {
        return speed;
    }

    public int getGear() {
        return gear;
    }

    private void updateGear() {
        if (speed >= 0 && speed <= 20) gear = 1;
        else if (speed >= 21 && speed <= 30) gear = 2;
        else if (speed >= 31 && speed <= 40) gear = 3;
        else gear = 4;
    }

    public void accelerate() {
        if (!isOn) return;

        speed += gear; // increment by current gear
        updateGear();
    }

    public void decelerate() {
        if (!isOn) return;

        speed -= gear; // decrement by current gear
        if (speed < 0) speed = 0;  // no negative speed
        updateGear();
    }
}
